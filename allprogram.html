public class FlagStarCoordinates
{
   
   /**
      * 计算五星红旗上星星顶点坐标的方法
      *
      * @param largeone 是否为大星，true表示大星，false表示小星
      * @param starOrder 小星的顺序（1 - 4），如果是大星则此参数无实际意义
      * @return 一个10 x 2的二维数组，代表10个顶点（5个外顶点和5个内顶点）的坐标，第一维表示顶点序号，第二维0表示x坐标，1表示y坐标
      */
   public static int[][] starCordinate(boolean largeone, int starOrder)
   {
      int width = 1152; // 设定旗宽像素
      int height = width * 2 / 3; // 根据长宽比3:2计算旗高像素
      
      // 大星外接圆直径（基于旗高）
      int largeDiameter = height * 3 / 10;
      // 小星外接圆直径（基于旗高）
      int smallDiameter = height * 1 / 10;
      
      // 大星圆心坐标（基于旗面的相对位置计算）
      int largeCenterX = width * 5 / 10;
      int largeCenterY = height * 5 / 10;
      
      // 小星圆心坐标数组（4颗小星的圆心坐标）
      int[][] smallCenters =
      {
         {
            width * 10 / 10, height * 2 / 10
         }
         ,
         {
            width * 12 / 10, height * 4 / 10
         }
         ,
         {
            width * 12 / 10, height * 7 / 10
         }
         ,
         {
            width * 10 / 10, height * 9 / 10
         }
      };
      
      int[][] coordinates = new int[10][2];
      if (largeone)
      {
         // 计算大星的顶点坐标
         int radius = largeDiameter / 2;
         double angleStep = 2 * Math.PI / 5;
         for (int i = 0; i < 5; i++)
         {
            double angle = i * angleStep;
            // 外顶点x坐标
            coordinates[i][0] = (int) (largeCenterX + radius * Math.cos(angle));
            // 外顶点y坐标
            coordinates[i][1] = (int) (largeCenterY - radius * Math.sin(angle));
            
            double innerAngle = angle + angleStep / 2;
            // 内顶点x坐标
            coordinates[i + 5][0] = (int) (largeCenterX + radius / 2 * Math.cos(innerAngle));
            // 内顶点y坐标
            coordinates[i + 5][1] = (int) (largeCenterY - radius / 2 * Math.sin(innerAngle));
         }
      }
      else
      {
         if (starOrder >= 1 && starOrder <= 4)
         {
            // 计算对应序号小星的顶点坐标
            int radius = smallDiameter / 2;
            int centerX = smallCenters[starOrder - 1][0];
            int centerY = smallCenters[starOrder - 1][1];
            double angleStep = 2 * Math.PI / 5;
            for (int i = 0; i < 5; i++)
            {
               double angle = i * angleStep;
               // 外顶点x坐标
               coordinates[i][0] = (int) (centerX + radius * Math.cos(angle));
               // 外顶点y坐标
               coordinates[i][1] = (int) (centerY - radius * Math.sin(angle));
               
               double innerAngle = angle + angleStep / 2;
               // 内顶点x坐标
               coordinates[i + 5][0] = (int) (centerX + radius / 2 * Math.cos(innerAngle));
               // 内顶点y坐标
               coordinates[i + 5][1] = (int) (centerY - radius / 2 * Math.sin(innerAngle));
            }
         }
      }
      return coordinates;
   }
}

public class DataAnalysis
{
   public static void main(String[] args) throws IOException
   {
      StringBuilder storestring = new StringBuilder();
      
      // Step 1: Initial pass to store data in StringBuilder
      Scanner s = new Scanner(System.in);
      int n = 0; // total row number
      String id, Grade, Score, Absence, Time, Start, Sub, Quiz, Order;
      
      // Accumulators for the column sums
      double idsum = 0, Gradesum = 0, Scoresum = 0, Absencesum = 0, Timesum = 0,
      Startsum = 0, Subsum = 0, Quizsum = 0, Ordersum = 0;

      Startmean = 0, Submean = 0, Quizmean = 0, Ordermean = 0;
      
      while (s.hasNext())
      {
         n++; // Increment the row counter
         id = s.next().trim();
         Grade = s.next().trim();
         Score = s.next().trim();
         Absence = s.next().trim();
         Time = s.next().trim();
         Start = s.next().trim();
         Sub = s.next().trim();
         Quiz = s.next().trim();
         Order = s.next().trim(); // Ensure Order is read properly
         
         storestring.append(id).append("\t");
         storestring.append(Grade).append("\t");
         storestring.append(Score).append("\t");
         storestring.append(Absence).append("\t");
         storestring.append(Time).append("\t");
         storestring.append(Start).append("\t");
         storestring.append(Sub).append("\t");
         storestring.append(Quiz).append("\t");
         storestring.append(Order).append("\n");
         
         if (n == 1) continue; // Skip the header row
            
         // Convert Grade to numeric value
         double GradeNum;
         if (Grade.equals("A")) GradeNum = 4;
            else if (Grade.equals("B")) GradeNum = 3;

                     Quizsum += Double.parseDouble(Quiz);

      System.out.println("Quizmean: "+ Quizmean);
      System.out.println("Ordermean: "+ Ordermean);
      
      // Step 2: Second pass to calculate standard deviations
      s = new Scanner(storestring.toString().trim());
      
      // Reset accumulators for standard deviation calculation
      idsum = 0; Gradesum = 0; Scoresum = 0; Absencesum = 0; Timesum = 0;
      Startsum = 0; Subsum = 0; Quizsum = 0; Ordersum = 0;
      
      double iddev, Gradedev, Scoredev, Absencedev, Timedev, Startdev, Subdev, Quizdev, Orderdev;
      
      while (s.hasNext())
      {
         id = s.next().trim();
         Grade = s.next().trim();
         Score = s.next().trim();
         Absence = s.next().trim();
         Time = s.next().trim();
         Start = s.next().trim();
         Sub = s.next().trim();
         Quiz = s.next().trim();
         Order = s.next().trim(); // Ensure Order is properly read
         
         // Calculate squared differences for standard deviation
         idsum += Math.pow(Double.parseDouble(id) - idmean, 2);
         double GradeNum;
         if (Grade.equals("A")) GradeNum = 4;
            else if (Grade.equals("B")) GradeNum = 3;
               else if (Grade.equals("C")) GradeNum = 2;
                  else if (Grade.equals("D")) GradeNum = 1;
                     else GradeNum = 0;
                        
                     Gradesum += Math.pow(GradeNum - Grademean, 2);
                     Scoresum += Math.pow(Double.parseDouble(Score) - Scoremean, 2);
                     Absencesum += Math.pow(Double.parseDouble(Absence) - Absencemean, 2);

                     Subsum += Math.pow(Double.parseDouble(Sub) - Submean, 2);
                     Quizsum += Math.pow(Double.parseDouble(Quiz) - Quizmean, 2);
                     Ordersum += Math.pow(Double.parseDouble(Order) - Ordermean, 2);
      }
      
      // Calculate standard deviations
      iddev = Math.sqrt(idsum / (n - 1));
      Gradedev = Math.sqrt(Gradesum / (n - 1));
      Scoredev = Math.sqrt(Scoresum / (n - 1));
      Absencedev = Math.sqrt(Absencesum / (n - 1));

      Subdev = Math.sqrt(Subsum / (n - 1));

   private int flagHeight;

      super(w);
      flagWidth = w;
      flagHeight = w * 2 / 3;
   }
   
   // Method to paint the moon (crescent)
   void paintMoon(Graphics g, Color bgcolor, int radius, int centerX, int centerY)
   {
      // Create the large circle (for the full moon)
      g.setColor(Color.WHITE);
      g.fillOval(centerX- radius/2, centerY - radius, 3* radius, 3* radius);
      
      // Create the small circle (to remove part of the moon to form the crescent)
      // The small circle should be positioned slightly to the right and also needs to be smaller than the large circle
      int smallCircleRadius = radius ; // Make the small circle smaller, to match the crescent shape
      int smallCircleX = centerX +radius / 6; // Move the small circle to the right
      int smallCircleY = centerY - smallCircleRadius; // Align it vertically to match the crescent shape
      
      g.setColor(bgcolor); // Fill with the background color (red in this case)
      g.fillOval(smallCircleX, smallCircleY, 2* smallCircleRadius, 2 * smallCircleRadius);
      
   }
   
   // Method to paint the star
   public void paintStar(double centerX, double centerY, double radius, double ang, Graphics g)
   {
      g.setColor(Color.WHITE); // White color for the star
      int[][] d = vertices(centerX, centerY, radius, ang);
      g.fillPolygon(d[0], d[1], 10);
   }
   
   // Paint method to draw the flag
   public void paint(Graphics g)
   {
      g.setColor(Color.RED); // Set the flag's background color
      g.fillRect(0, 0, flagWidth, flagHeight); // Fill the background with red
      
      // Constants for the moon and star
      int moonRadius = flagHeight / 6;
      int moonCenterX = flagWidth / 4; // Move moon to the left side
      int moonCenterY = flagHeight / 2;
      int starRadius = flagHeight / 12;
      double starCenterX = flagWidth / 2.4; // Position star to the right of the moon
      double starCenterY = flagHeight / 2;
      double starAngle = Math.toRadians(72); // Angle for the star's points
      
      
      paintMoon(g, Color.RED, moonRadius, moonCenterX, moonCenterY);
      
      // Paint the star
      paintStar(starCenterX, starCenterY, starRadius, starAngle, g);
   }

   }

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import java.io.*;
import java.util.*;

public class Datasheet extends JFrame
{
   // Instance variables for the file path and number of rows
   private String filepath;
   private int totalRows;
   
   // Constructor
   public Datasheet(String filepath, int totalRows)
   {
      this.filepath = filepath;
      this.totalRows = totalRows;
      initializeUI();
   }
   
   // Initialize the UI components
   private void initializeUI()
   {
      setTitle("Editable Datasheet");
      setSize(800, 600);
      setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      
      // Set layout to GridLayout or any other suitable layout for rows and columns
      setLayout(new GridLayout(totalRows + 1, 9)); // totalRows + 1 for header row
      
      // Create the menu bar with a Save item
      JMenuBar menuBar = new JMenuBar();
      JMenu fileMenu = new JMenu("File");
      JMenuItem saveItem = new JMenuItem("Save");
      
      // Add action listener to Save item
      saveItem.addActionListener(new ActionListener()
      {
         @Override
         public void actionPerformed(ActionEvent e)
         {
            saveData();
         }
      }
      );
      
      fileMenu.add(saveItem);
      menuBar.add(fileMenu);
      setJMenuBar(menuBar);
      
      // Create the header row (JLabel)
      String[] headers =
      {
         "ID", "Grade", "Score", "Age", "Name", "Gender", "Course", "Semester", "Teacher"
      };
      for (int col = 0; col < 9; col++)
      {
         JLabel headerLabel = new JLabel(headers[col]);
         headerLabel.setName("num0_"+ col); // Set name for header
         add(headerLabel);
      }
      
      // Create editable rows
      for (int row = 1; row <= totalRows; row++)
      {
         for (int col = 0; col < 9; col++)
         {
            Component cell = null;
            if (col == 1)
            {
               // JComboBox for column 1 (Grade)
               JComboBox<String> cmb = new JComboBox<>(new String[]
               {
                  "A", "B", "C", "D", "F"
               }
               );
               cmb.setName("num"+ row + "_"+ col); // Set name for the combo box
               cell = cmb;
            }
            else
            {
               // JTextField for other columns
               JTextField txt = new JTextField();
               txt.setName("num"+ row + "_"+ col); // Set name for the text field
               cell = txt;
            }
            add(cell);
         }
      }
   }
   
   // Method to find a component at a specific row and column
   private Component get(int row, int col)
   {
      Component[] components = getContentPane().getComponents();
      for (Component comp : components)
      {
         if (comp.getName().equals("num"+ row + "_"+ col))
         {
            return comp;
         }
      }
      return null; // If not found
      
   }
   
   // Method to save the data to a file
   private void saveData()
   {
      StringBuilder stringBuilder = new StringBuilder();
      try
      {
         // Loop through all rows and columns to get the values
         for (int row = 1; row <= totalRows; row++)
         {
            for (int col = 0; col < 9; col++)
            {
               Component comp = get(row, col);
               String value = "";
               
               // Check if the component is a JTextField or JComboBox
               if (comp instanceof JTextField)
               {
                  value = ((JTextField) comp).getText();
               }
               else if (comp instanceof JComboBox)
               {
                  value = (String) ((JComboBox<?>) comp).getSelectedItem();
               }
               
               // Append value to StringBuilder and separate with commas
               stringBuilder.append(value);
               if (col < 8)
               {
                  stringBuilder.append(","); // Use comma as delimiter
                  
               }
            }
            stringBuilder.append("\n"); // New line after each row
            
         }
         
         // Write the data to a file
         FileWriter writer = new FileWriter(filepath + "1");
         writer.write(stringBuilder.toString());
         writer.close();
         JOptionPane.showMessageDialog(this, "Data saved successfully!");
      }
      catch (IOException e)
      {
         JOptionPane.showMessageDialog(this, "Error saving data: "+ e.getMessage());
      }
   }
   
   // Main method to launch the application
   public static void main(String[] args)
   {
      // Set the file path and number of rows (these can be dynamic as per requirements)
      String filepath = "path/to/your/file"; // Set your actual file path
      int totalRows = 10; // Set the number of rows as needed
      
      // Create and show the Datasheet window
      SwingUtilities.invokeLater(new Runnable()
      {
         @Override
         public void run()
         {
            new Datasheet(filepath, totalRows).setVisible(true);
         }

import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.HashSet;
import java.util.Set;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

public class GogameServer
{
   private final Map<String, Set<Integer>> stones = new ConcurrentHashMap<>();
   private final int N = 19;
   private final ExecutorService pool = Executors.newFixedThreadPool(10);
   private final int port = 2311;
   
   class Worker implements Runnable
   {
      private final Socket socket;
      
      public Worker(Socket s)
      {
         this.socket = s;
      }
      
      @Override
      public void run()
      {
         String ip = socket.getInetAddress().getHostAddress(); // 去掉斜杠
         try (DataInputStream d = new DataInputStream(socket.getInputStream()))
         {
            Set<Integer> coordinates = stones.computeIfAbsent(ip, k -> new HashSet<>());
            int r = d.readInt();
            int c = d.readInt();
            coordinates.add(r * 100 + c);
            
            double distance = 2 * N;
            int minr = 0, minc = 0;
            for (int row = 0; row < N; row++)
            {
               for (int col = 0; col < N; col++)
               {
                  if (coordinates.contains(row * 100 + col)) continue;
                     double dist = Math.sqrt((col - c) * (col - c) + (row - r) * (row - r));
                  if (dist < distance)
                  {
                     distance = dist;
                     minr = row;
                     minc = col;
                  }
               }
            }
            coordinates.add(minr * 100 + minc);
            sendnum(ip, minr, minc);
         }
         catch (Exception e)
         {
            e.printStackTrace(); // 至少打印堆栈跟踪
            
         }
         finally
         {
            try
            {
               socket.close();
            }
            catch (Exception ignore)
            {
               
            }
         }
      }
   }
   
   protected void sendnum(String ip, int r, int c)
   {
      // 客户端应该有一个不同的端口来接收响应
      int clientResponsePort = 2312; // 假设这是客户端用来接收响应的端口
      new Thread(() ->
      {
         try (Socket s = new Socket(ip, clientResponsePort);
         DataOutputStream o = new DataOutputStream(s.getOutputStream()))
         {
            o.writeInt(r);
            o.writeInt(c);
         }
         catch (Exception e)
         {
            e.printStackTrace(); // 至少打印堆栈跟踪
            
         }
      }
      ).start();
   }
   
   public void receive()
   {
      try (ServerSocket ss = new ServerSocket(port))
      {
         while (true)
         {
            Socket s = ss.accept();
            Worker w = new Worker(s);
            pool.execute(w);
         }
      }
      catch (Exception e)
      {
         e.printStackTrace(); // 至少打印堆栈跟踪
         
      }
   }
   
   public static void main(String[] args)
   {
      new GogameServer().receive();
   }
}
